CREATE TABLE TB_PERSONA(
	VAR_ID_PERSONA VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_PERSONA	 VARCHAR2(30),	
	VAR_TEL_PERSONA VARCHAR2(10),
	VAR_GENERO_PERSONA VARCHAR2(10),
	DT_FECHA_NAC_PERSONA DATE DEFAULT SYSDATE,
	VAR_CORREO_PERSONA VARCHAR2(50),
	VAR_PROVINCIA_PERSONA VARCHAR(30),
	VAR_CANTON_PERSONA VARCHAR(30),
	VAR_PARROQUIA_PERSONA VARHCAR(30)
);
CREATE TABLE TB_HIJO(
	VAR_ID_HIJO VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_HIJO VARCHAR2(30),
	INT_EDAD_HIJO INTEGER NOT NULL,
	VAR_ID_PERSONA VARCHAR2(50),
    CONSTRAINT FK_PERSONA
	FOREIGN KEY (VAR_ID_PERSONA) REFERENCES TB_PERSONA(VAR_ID_PERSONA)
);
CREATE TABLE TB_AFILIACION(
	VAR_ID_AFILIACION VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_AFILIACION VARCHAR2(20),
	VAR_TIPO_AFILIACION VARCHAR2(20),
	VAR_ESTADO_AFILIACION VARCHAR2(10)
);

CREATE TABLE TB_BASELEGAL(
	VAR_ID_BASELEGAL VARCHAR2(50) PRIMARY KEY,
	VAR_DESCRIPCION_BASELEGAL VARCHAR2(50),
	VAR_DECRETO_BASELEGAL VARCHAR2(80),
	VAR_ESTADO_BASELEGAL VARCHAR2(10),
	VAR_ID_AFILIACION VARCHAR2(50),
	CONSTRAINT FK_AFILIACION
	FOREIGN KEY (VAR_ID_AFILIACION) REFERENCES TB_AFILIACION(VAR_ID_AFILIACION)
);
CREATE TABLE TB_CODIGO(
	VAR_ID_CODIGO VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_CODIGO VARCHAR2(30),
	INT_ARTICULO INTEGER NOT NULL,
	VAR_ESTADO_CODIGO VARCHAR2(10),
	VAR_DETALLE_CODIGO VARCHAR2(30),
	VAR_ID_BASELEGAL VARCHAR2(50),
	CONSTRAINT FK_BASELEGAL
	FOREIGN KEY (VAR_ID_BASELEGAL) REFERENCES TB_BASELEGAL(VAR_ID_BASELEGAL)
);
CREATE TABLE TB_APORTACION(
	VAR_ID_APORTACION VARCHAR2(50) PRIMARY KEY,
	DT_FECHA_APORTACION DATE DEFAULT SYSDATE,
	INT_NO_APORTACION INTEGER NOT NULL,
	VAR_DETALLE_APORTACION VARCHAR2(50),
	VAR_MONTO_APORTACION VARCHAR2(10)
);

CREATE TABLE TB_EMPRESA(
	VAR_ID_EMPRESA VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_EMPRESA VARCHAR2(40),
	VAR_DETALLE_EMPRESA VARCHAR2(40),
	VAR_PROVINCIA_PERSONA VARCHAR(30),
	VAR_CANTON_PERSONA VARCHAR(30),
	VAR_PARROQUIA_PERSONA VARHCAR(30),
	VAR_RESPONSABLE_EMPRESA VARCHAR2(30)
);
CREATE TABLE TB_USUARIO(
	VAR_ID_USUARIO VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_USUARIO VARCHAR2(50),
	VAR_CONTRASENIA_USUARIO VARCHAR2(50),
	VAR_ID_PERSONA VARCHAR2(50),
    CONSTRAINT FK_ID_PERSONA
	FOREIGN KEY (VAR_ID_PERSONA) REFERENCES TB_PERSONA(VAR_ID_PERSONA)
);


CREATE TABLE TB_PERFIL(
	VAR_ID_PERFIL VARCHAR2(50) PRIMARY KEY,
	VAR_DESCRIPCION_PERFIL VARCHAR2(50),
	VAR_NOMBRE_PERFIL VARCHAR2(30),
	VAR_ID_USUARIO VARCHAR2(50),
	CONSTRAINT FK_USUARIO
	FOREIGN KEY (VAR_ID_USUARIO) REFERENCES TB_USUARIO(VAR_ID_USUARIO)
);

CREATE TABLE TB_ROL(
	VAR_COD_ROL VARCHAR2(50) PRIMARY KEY,
	VAR_NOMBRE_ROL VARCHAR2(20),
	VAR_DETALLE_ROL VARCHAR2(20),
	VAR_ID_PERFIL VARCHAR(50),
	CONSTRAINT FK_PERFIL
	FOREIGN KEY (VAR_ID_PERFIL) REFERENCES TB_PERFIL(VAR_ID_PERFIL)
);
CREATE TABLE TB_TRABAJO( 
	VAR_ID_TRABAJO VARCHAR2(50) PRIMARY KEY,
	VAR_ID_PERSONA VARCHAR2(50),
	VAR_ID_EMPRESA VARCHAR2(50),
	VAR_ID_APORTACION VARCHAR2(50),
	VAR_ID_AFILIACION VARCHAR2(50),
	FOREIGN KEY (VAR_ID_PERSONA) REFERENCES TB_PERSONA(VAR_ID_PERSONA),
	FOREIGN KEY (VAR_ID_EMPRESA) REFERENCES TB_EMPRESA(VAR_ID_EMPRESA),
	FOREIGN KEY (VAR_ID_APORTACION) REFERENCES TB_APORTACION(VAR_ID_APORTACION),
	FOREIGN KEY (VAR_ID_AFILIACION) REFERENCES TB_AFILIACION(VAR_ID_AFILIACION)
);
